## 异步控制并发数 ##

写一个函数，可以控制最大并发数。
例如最大并发10个，有一个成功后再追加一个请求。
```javascript
let time = 1
function fetch2() {
	if(time>3){
		time = 1
	}
	return new Promise(function(resolve) {
		setTimeout(() => {
			resolve()
		}, 200 * time++)
	})
}
function promiseAll(urls, num) {
    this.res = [];
    this.urls = urls;
    this.len = urls.length;
    this.curList = this.urls.splice(0, num);
    this.count = 0;
}

promiseAll.prototype.runTask = function() {
    return new Promise((resolve, reject) => {
        this.loop(resolve, reject);
    })
}
promiseAll.prototype.loop = function(resolve, reject) {
    while(this.curList.length) {
        const url = this.curList.shift();
        fetch2(url).then(res => {
            this.count++;
            this.res.push(url);
            if(this.urls.length) {
                this.curList.push(...this.urls.splice(0, 1));
                this.loop(resolve, reject);
			}
            if(this.count === this.len) {
                resolve(this.res);
            }
        }).catch(err => {
            reject(err);
        })
    }
}

const promiseAllInstance = new promiseAll([11,22,33,44,55,66,77,88], 4)
promiseAllInstance.runTask().then(res=> {
    console.log(res)
})


function mutilFetch(urls, max) {
    let fetchSum = urls.length;
    let curUrls = urls.splice(0, max);
    let endFetchSum = 0;
    let fetchRes = [];

    const loop = (resolve, reject) => {
        while (curUrls.length) {
            const url = curUrls.shift();
            fetch2(url).then(res => {
                endFetchSum++;
                fetchRes.push(res);

                if (urls.length) {
                    curUrls.push(...urls.splice(0, 1));
                    loop(resolve, reject);
                }
                if (endFetchSum === fetchSum) {
                    resolve(fetchRes);
                }
            });
        }
    }
    
    return new Promise((resolve, reject) => {
        loop(resolve, reject);
    })
}

mutilFetch([11,22,33,44,55,66,77,88], 4).then(res => {
    console.log(res)
})
```
