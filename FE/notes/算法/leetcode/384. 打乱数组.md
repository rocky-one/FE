## 打乱数组 ##


```javascript
// 取0-N之间的随机数方法
function getRandom(min, max) {
	return Math.floor(Math.random() * (max - min)) + min
}
// 遍历数组 根据随机数交互数组位置
function shuffle(arr) {
	for(let i = 0; i < arr.length; i++) {
		let randNum = getRandom(i, arr.length - 1)
		let item = arr[i]
		arr[i] = arr[randNum]
		arr[randNum] = item
	}
	return arr
}
console.log(shuffle([1,2,3,4,5,6,7,8]))
```
