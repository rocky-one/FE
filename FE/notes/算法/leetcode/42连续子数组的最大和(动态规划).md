## 连续子数组的最大和 ##

```javascript
// 动态规划填表
function maxSubArray(nums) {
  const len = nums.length
  if(len === 1) return nums[0]
  const dp = []
  for(let i = 0; i < len; i++) {
    dp.push(new Array(len).fill(false))
  }
  // 初始化第一项 注意要判断是否是最大值
  dp[0][0] = nums[0]
  let max = -101
  if(dp[0][0] > max) {
    max = dp[0][0]
  }
  // 先填写对角线，后续只需要遍历对角线的一半即可
  for(let i = 1; i < len; i++) {
    // 对角线
    dp[i][i] = nums[i]
    // 顺便把第一行也计算
    dp[0][i] = dp[0][i-1] + nums[i]
    if(dp[i][i] > max) {
      max = dp[i][i]
    }
    if(dp[0][i] > max){
      max = dp[0][i]
    }
  }
  // 从第二行开始，第一行已经初始化过
  for(let i = 1; i < len; i++) {
    // 从对角线的位置的后一个开始，因为对角线已经计算过
    for(let j = i + 1; j < len; j++) {
      // 状态转移，当前位置=前一个位置+当前值
      // 所以前一个要有值，前一个就是对角线那个值
      dp[i][j] = dp[i][j-1] + nums[j]
      if(dp[i][j] > max) {
        max = dp[i][j]
      }
    }
  }
  return max
}
let nums = [-2,1,-3,4,-1,2,1,-5,4]
console.log(maxSubArray(nums)) // 6
```
