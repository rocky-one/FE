## 插入排序 ##


	function insertionSort(arr) {
		for (let i = 1; i < arr.length; i++) {
			let temp = arr[i]
			let pre = i - 1
			while (pre >= 0 && arr[pre] > temp) {
				arr[pre + 1] = arr[pre]
				pre--
			}
			arr[pre + 1] = temp
		}
		return arr
	}

	const arr = [5, 3, 1, 2, 9, 6, 4]
	console.log(insertionSort(arr))
