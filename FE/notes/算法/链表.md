* 创建链表
```javascript
function CreateNode(element) {
  this.element = element
  this.next = null
}
function insertNextNode(targetNode, node) {
  targetNode.next = node;
}
function createLinkList(list){
  let root = {
    name: 'root'
  }
  let targetNode = root
  list.forEach(item => {
    const nextNode = new CreateNode(item)
    insertNextNode(targetNode, nextNode)
    targetNode = nextNode
  })
  return root
}
const list = [
  {
    name: 1
  },
  {
    name: 2
  },
  {
    name: 3
  },
  {
    name: 4
  }
]
const linkList2 = createLinkList(list)
```

* 反转链表
```javascript
function reverseLink(head) {
  if(!head || !head.next) return head
  let cur = head
  let pre = null
  while(cur){
    const curNext = cur.next
    // 将下一个的指针指向前一个
    // root指向null，1指向root，2指向1，3指向2
    cur.next = pre
    pre = cur
    cur = curNext
  }
  return pre
}
```

* 反转分组链表
```javascript
function reverseLink(link){
  let pre = null
  let cur = link
  while(cur){
    let next = cur.next
    cur.next = pre
    pre = cur
    cur = next
  }
  return pre
}
function reverseLinkGroup(link, group){
  let pre = link
  let end = link
  while(end.next !== null) {
    for(let i = 0; i < group && end !== null; i++){
      end = end.next
    }
    if(end === null) break
    let start = pre.next
    // 先把上一组最后一个节点的next指针保存一下，等到下一组开始时
    let next = end.next
    end.next = null
    pre.next = reverseLink(start)
    // start此时已经置换到当前组的最后一个，把刚刚存的next.next 给 start.next
    start.next = next
    // 更新上一个为当前组最后一个
    pre = start
    // 更新结束节点 从当前节点重新开始计算
    end = pre
  }
  return link
}
```

* 判断是否有环形链表
利用快慢指针，快指针先于慢指针两个节点，当存在环状指针是两个快慢指针一定会相遇，当相遇时就结束。
```javascript
function linkHasLoop(linkList2){
  if(!linkList2) return false
  let cur = linkList2
  let fast = linkList2.next
  while(fast && fast.next){
    cur = cur.next
    fast = fast.next.next
    if(fast === null){
      return false
    }else if(cur === fast){
      return true
    }
  }
  return false
}
function linkHasLoop2(linkList2){
  if(!linkList2) return false
  let cur = linkList2
  let fast = linkList2.next
  while(cur !== fast){
    if(!fast || !fast.next){
      return false
    }
    cur = cur.next
    fast = fast.next.next
  }
  return true
}
```

* 合并两个有序链表
利用递归
```javascript
function mergeTwoLinks(l1, l2){
  if(l1 === null) {
    return l2
  }else if(l2 === null) {
    return l1
  }else if(l1.element.val < l2.element.val){
    l1.next = mergeTwoLinks(l1.next, l2)
    return l1
  }else {
    l2.next = mergeTwoLinks(l1, l2.next)
    return l2
  }
}
```

* 移除倒数第n个节点
利用双指针，第一个first从n+1的位置开始，第二个second从开始位置开始，这样两个之间永远保持间隔n，这样当第一个节点到达最后的时候，第二个节点和第一个节点保持n个间隔，也就是倒数第n个的位置。
```javascript
function removeNthFromEnd(link, n){
  let first = link
  let second = link
  for(let i = 0; i <= n+1; i++){
    first = first.next
  }
  while(first!==null){
    first = first.next
    second = second.next
  }
  second.next = second.next.next
  return link
}
```

* 求链表的中间节点
```javascript
// 快慢指针慢指针正常，快指针是慢指针的2倍，当快指针到达尾部时慢指针的位置刚好是中间位置。
function getMiddleNode(link){
  let fast = link
  let slow = link
  while(fast !== null && fast.next !== null) {
    fast = fast.next.next
    slow = slow.next
  }
  return slow
}
```
